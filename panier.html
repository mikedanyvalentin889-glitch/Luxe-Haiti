function renderCart() { const cart = getPanier(); const itemsContainer = document.getElementById('cartItems'); const emptyState = document.getElementById('cartEmpty'); const layout = document.getElementById('cartLayout'); if (cart.length === 0) { layout.style.display = 'none'; emptyState.style.display = 'flex'; return; } layout.style.display = 'grid'; emptyState.style.display = 'none'; const grouped = cart.reduce((acc, item) => { acc[item.id] = (acc[item.id] || 0) + 1; return acc; }, {}); let html = ''; let subtotal = 0; for (const [id, qty] of Object.entries(grouped)) { const product = produits.find(p => p.id == id); if (!product) continue; const itemTotal = product.prix * qty; subtotal += itemTotal; html += `

${product.nom}

${product.categorie === 'vetements' ? 'Vêtements' : 'Mécanique'}

${product.prix.toLocaleString()} HTG
${qty}
${itemTotal.toLocaleString()} HTG
`; } itemsContainer.innerHTML = html; document.getElementById('subtotal').textContent = subtotal.toLocaleString() + ' HTG'; document.getElementById('total').textContent = subtotal.toLocaleString() + ' HTG'; updateCartCount(); }
